def get_graph():
    g = pgv.AGraph(
        size="16,9", ratio="fill", resolution="300", rankdir="LR", directed="true",
    )

    g.node_attr["width"] = "1.0"
    g.node_attr["fontsize"] = "35"
    g.edge_attr["penwidth"] = "1"
    g.node_attr["style"] = "filled"
    g.node_attr["fillcolor"] = "black"
    return g


def wrapping_up(g, fname):
    g.draw(fname, prog="dot")
    return Image(fname)

# graph with bidirected, dashed, curved edges
def get_graph_bdc():
    g = pgv.AGraph(
        size="16,9",
        ratio="fill",
        resolution="300",
        rankdir="LR",
        directed="true",
        splines="curved",
    )

    g.node_attr["width"] = "1.0"
    g.node_attr["fontsize"] = "35"
    g.edge_attr["penwidth"] = "1"
    g.node_attr["style"] = "filled"
    g.node_attr["fillcolor"] = "black"
    g.edge_attr["style"] = "dashed"
    g.edge_attr["dir"] = "both"
    return g
